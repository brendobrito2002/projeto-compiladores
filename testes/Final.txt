programa Final;
{
    var x: inteiro;
    var y: booleano;
    var z: inteiro;
    var a: inteiro;
    var b: inteiro;
    var temp: inteiro;
    a = 10;

    funcao calcula(a: inteiro, b: inteiro): inteiro {
        se (a > b) {
            temp = a + b;
            retorno temp;
        } senao {
            temp = a * b;
            retorno temp;
        }
    }

    var valor: inteiro;

    funcao analisaValor(valor: inteiro): booleano {
        se (valor > 5) {
            retorno true;
        } senao {
            retorno false;
        }
    }

    var n: inteiro;
    var limite: inteiro;

    funcao calculaFatorialComLimite(n: inteiro, limite: inteiro): inteiro {
        var resultado: inteiro;
        resultado = 1;

        enquanto (n > 1) {
            se (resultado > limite) {
                print(resultado);
                break;
            }

            resultado = resultado * n;
            n = n - 1;
        }

        retorno resultado;
    }

    funcao processaValores(a: inteiro, b: inteiro): inteiro {
        var resultado: inteiro;

        se (a > b) {
            resultado = calcula(a, b);
        } senao {
            resultado = calcula(b, a);
        }

        enquanto (resultado < 50) {
            resultado = resultado + 5;

            se (resultado == 30) {
                print(resultado);
                continue;
            }

            se (resultado > 40) {
                break;
            }

            print(resultado);
        }

        retorno resultado;
    }

    x = calcula(8, 3);
    y = analisaValor(x);
    z = calculaFatorialComLimite(5, 60);

    se (y == true) {
        print(x);
        print(z);
    } senao {
        print(0);
    }

    z = processaValores(x, z);

    enquanto (z < 100) {
        z = z + 7;
        print(z);

        se (z > 90) {
            break;
        }
    }
}
